is_alone_or_in_faction_at_peace_with_CAL = {
	OR = {
		is_in_faction = no
		is_in_faction_with = CAL
		AND = {
			is_in_faction = yes
			NOT = {
				is_in_faction_with = CAL
			}
		}
	}
	NOT = {
		any_allied_country = {
			has_war_with = CAL
		}
	}
}
